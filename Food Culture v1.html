def agewiseanalysis(labelname):
    if (labelname == 'Age30'):
        label = ds.loc[ ( ds['Age'] >= 30 ) ]
    if (labelname == 'Age25to30'):
        label =  ds.loc[ ( ds['Age'] >= 25 ) & ( ds['Age'] < 30 ) ]
    if (labelname == 'Age20to25'):
        label =  ds.loc[ ( ds['Age'] >= 20 ) & ( ds['Age'] < 25 ) ]
    if (labelname == 'Age20'):
        label = ds.loc[ ( ds['Age'] < 20 ) ]
    print('Number of entries',len(label.columns))
    print("City names and Number of entries for age group " , labelname)
    city_frame = pd.DataFrame(data = {'Names': label.City.unique() ,
                                      'NumEntries': [label.City.value_counts()[str(data)] for data in label.City.unique()] })
    city_frame['Percentage:'] = [data/city_frame['NumEntries'].sum()*100 for data in city_frame['NumEntries']]
    print(city_frame.sort_values('NumEntries', ascending=False).to_string(index=False))

    #Employment Status:
    Employmentstatus = pd.DataFrame(data = {'Names': label.CurrentEmploymentStatus.unique() ,
